    %%%%%%%%%%%%%%%%%%%%%
    % This set of commands places a
    % number at the location of \detail
    % with the intention that that is 
    % somplace students should then 
    % fill in details.
    %%%%%%%%%%%%%%%%%%%%%%
    \newcounter{DetailsOn}
    \setcounter{DetailsOn}{0}
    \newcommand{\DetailsOn}{
        \setcounter{DetailsOn}{1}    
    }
    \newcommand{\DetailsOff}{
        \setcounter{DetailsOn}{0}    
    }
    \newcounter{detail}
    \setcounter{detail}{1}
    \newcommand{\detail}[1][secondaryaccent]{
        \ifthenelse{\value{DetailsOn}=1}{
            \textcolor{#1}{
                \relax\ifmmode 
                    ^{\mathbf{\left(\thedetail\right)}}
                \else
                    \scriptsize
                    \hspace{-8pt}\raisebox{0.75\baselineskip}{\textbf{(\thedetail)}}\hspace*{-8pt}
                \fi
            }
            \stepcounter{detail}
        }{}
    }
